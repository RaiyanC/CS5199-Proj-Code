502##
503#F  FactorsRho( <n>, <inc>, <cluster>, <limit> )   Pollards rho factorization
504##
505##  `FactorsInt' does trial divisions by the primes less than 1000 to  detect
506##  all composites with a factor less than 1000 and primes less than 1000000.
507##  After that it calls `FactorsRho(<n>,1,16,8192)' to do the hard work.
508##
509##  `FactorsRho'  will  return a  list  of factors   and a list  of composite
510##  number.   Usually  `FactorsInt'  factors  integers  with   prime  factors
511##  $\<1000$ faster.     However  for   integers  with  no   factor  $\<1000$
512##  `FactorsRho' will be faster.
513##
514##  `FactorsRho' uses Pollards $\rho$ method to factor the integer $n = p q$.
515##  For a small simple example lets assume we want to factor $667 = 23 * 29$.
516##  `FactorsRho' first calls `IsPrimeInt' to avoid trying to factor a prime.
517##
518##  Then it uses the sequence defined by  $x_0=1, x_{i+1}=(x_i^2+1)$ mod $n$.
519##  In our example this is $1, 2, 5, 26, 10, 101, 197, 124, 36, 630, .. $.
520##
521##  Modulo $p$ it takes on at most $p-1$ different values, thus it eventually
522##  becomes recurrent, usually this happens after roughly $2 \sqrt{p}$ steps.
523##  In our example modulo 23 we get $1, 2, 5, 3, 10, 9, 13, 9, 13, 9, .. $.
524##
525##  Thus there exist pairs $i, j$ such that $x_i = x_j$ mod $p$,  i.e.,  such
526##  that $p$ divides $Gcd( n, x_j-x_i )$.  With a bit of luck no other factor
527##  of $n$ divides $x_j - x_i$ so we find $p$ if we know such a pair.  In our
528##  example $5, 7$ is the first pair, $x_7-x_5=23$, and $Gcd(667,23) = 23$.
529##
530##  Now it is too expensive to check all pairs, but there also must be  pairs
531##  of the form $2^i-1, j$ with $3*2^{i-1} <= j < 4*2^{i-1}$.  In our example
532##  $7, 13$ is the first such pair, $x_13-x_7=506$, and $Gcd(667,506) = 23$.
533##
534##  Thus by taking the gcds of $n$ and $x_j-x_i$ for such pairs, we will find
535##  the factor $p$ after approximately $2 \sqrt{p} \<= 2 \sqrt^4{n}$ steps.
536##
537##  If $Gcd( n, x_j - x_i )$  is not a prime `FactorsRho'  will  call  itself
538##  recursively with a different value for <inc>, i.e., it will try to factor
539##  the gcd using a different sequence $x_{i+1} = (x_i^2 + inc)$ mod $n$.
540##
541##  Since the gcd computations are by far the most time consuming part of the
542##  algorithm  one can save time by  clustering differences and computing the
543##  gcd  only every <cluster>  iteration.  This slightly increases the chance
544##  that a gcd is composite, but reduces the runtime by a large amount.
545##
546##  Finally `FactorsRho' accepts an argument <limit>  which is the number  of
547##  iterations  performed by `FactorsRho' before giving up. The default value
548##  is  8192  which corresponds to a few minutes  while guaranteeing that all
549##  prime factors less than $10^6$ and most less than $10^9$ are found.
550##
551##  Better descriptions of the algorithm and related topics can be found  in:
552##  J. Pollard, A Monte Carlo Method for Factorization, BIT 15, 1975, 331-334
553##  R. Brent, An Improved Monte Carlo Method for Fact., BIT 20, 1980, 176-184
554##  D. Knuth, Seminumerical Algorithms  (TACP II),  AddiWesl,  1973,  369-371
555##
556FactorsRho := function ( n, inc, cluster, limit )
557
558    local   i,  sign,  factors,  composite,  x,  y,  k,  z,  g,  tmp,
559            IsPrimeOrProbablyPrimeInt;
560
561    # make $n$ positive and handle trivial cases
562    sign := 1;
563    if n < 0  then sign := -sign;  n := -n;  fi;
564    if n < 4  then return [ [ sign * n ], [] ];  fi;
565    factors   := [];
566    composite := [];
567    while n mod 2 = 0  do Add( factors, 2 );  n := n / 2;  od;
568    while n mod 3 = 0  do Add( factors, 3 );  n := n / 3;  od;
569
570    if   ValueOption("UseProbabilisticPrimalityTest") = true
571    then IsPrimeOrProbablyPrimeInt := IsProbablyPrimeInt;
572    else IsPrimeOrProbablyPrimeInt := IsPrimeInt; fi;
573
574    if IsPrimeOrProbablyPrimeInt(n)  then Add( factors, n );  n := 1;  fi;
575
576    # initialize $x_0$
577    x := 1;  z := 1;  i := 0;
578
579    # loop until we have factored $n$ completely or run out of patience
580    while 1 < n  and 2^i <= limit  do
581
582        # $y = x_{2^i-1}$
583        y := x;  i := i + 1;
584
585        # $x_{2^i}, .., x_{3*2^{i-1}-1}$ need not be compared to $x_{2^i-1}$
586        for k  in [1..2^(i-1)]  do
587            x := (x^2 + inc) mod n;
588        od;
589
590        # compare $x_{3*2^{i-1}}, .., x_{4*2^{i-1}-1}$ with $x_{2^i-1}$
591        for k  in [1..2^(i-1)]  do
592            x := (x^2 + inc) mod n;

595            # from time to time compute the gcd
596            if k mod cluster = 0  then
597                g := GcdInt( n, z );
598
599                # if it is > 1 we have found a factor which need not be prime
600                if g > 1  then
601                    tmp := FactorsRho(g,inc+1,QuoInt(cluster+1,2),limit);
602                    factors   := Concatenation( factors,   tmp[1] );
603                    composite := Concatenation( composite, tmp[2] );
604
605                    n := n / g;
606                    if IsPrimeOrProbablyPrimeInt(n)  then
607                        Add( factors, n );  n := 1;
608                    fi;
609                fi;
610            fi;
611        od;
612    od;
613
614    # add <n> to the list of composite numbers
615    if 1 < n  then
616        Add( composite, n );
617    fi;
618
619    # sort the list of factors and composite numbers and return it
620    Sort(factors);
621    Sort(composite);
622    if 0 < Length(factors)  then
623        factors[1] := sign * factors[1];
624    else
625        composite[1] := sign * composite[1];
626    fi;
