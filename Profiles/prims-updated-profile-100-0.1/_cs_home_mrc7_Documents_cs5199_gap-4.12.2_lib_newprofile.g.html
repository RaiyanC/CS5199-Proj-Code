18##  <#GAPDoc Label="ProfileLineByLine">
19##  <ManSection>
20##  <Func Name="ProfileLineByLine" Arg="filename[,options]"/>
21##
22##  <Description>
23##  <Ref Func="ProfileLineByLine"/> begins GAP recording profiling
24##  data to the file <A>filename</A>. This file will get *very* large
25##  very quickly. This file is compressed using gzip to reduce its size.
26##
27##  <A>options</A> is an optional dictionary, which sets various
28##  configuration options. These are
29##  <List>
30##  <Mark>coverage</Mark>
31##      <Item> Boolean (defaults to false). If this is enabled,
32##      only information about which lines are read and executed is
33##      stored. Enabling this is the same as calling
34##      <Ref Func="CoverageLineByLine"/>. Using this ignores all other
35##      options.</Item>
36##
37##  <Mark>wallTime</Mark>
38##      <Item> Boolean (defaults to true). Sets if time should be measured
39##             using wall-clock time (true) or CPU time (false).
40##             (measuring CPU-time has a higher overhead).
41##      </Item>
42##
43##  <Mark>recordMem</Mark>
44##      <Item> Boolean (defaults to false). Instead of recording the
45##             CPU time taken by statements, record the total size of all
46##             new objects created by each line.
47##      </Item>
48##
49##  <Mark>resolution</Mark>
50##      <Item> Integer (defaults to 0). By default profiling will record a trace
51##             of all executed code. When <A>resolution</A> non-zero, GAP
52##             instead samples which piece of code is being executed every
53##             <A>resolution</A> nanoseconds. Increasing this
54##             improves performance and produces smaller traces,
55##             at the cost of accuracy. GAP will still accurately record
56##             which statements are executed at least once.</Item>
57##  </List>
58##  </Description>
59##  </ManSection>
60##  <#/GAPDoc>
61##
62BIND_GLOBAL("ProfileLineByLine",function(arg)
63    local optRec, r;   
64    if Length(arg) = 0 or Length(arg) > 2 or not(IsString(arg[1])) then
65      Error("usage: ProfileLineByLine(filename,[options]");
66    fi;
67
68    optRec := rec(coverage := false,
69                  wallTime := true,
70                  recordMem := false,
71                  resolution := 0);
72    if Length(arg) = 2 then
73      if not(IsRecord(arg[2])) then
74        Error("usage: ProfileLineByLine(filename,[options]");
75      fi;
76      
77      for r in RecNames(arg[2]) do
78        if not(IsBound(optRec.(r))) then
79          Error(Concatenation("Bad option: ", r));
80        fi;
81        optRec.(r) := arg[2].(r);
82      od;
83    fi;
84    
85    if Length(arg[1]) < 3 or 
86       arg[1]{[Length(arg[1])-2..Length(arg[1])]} <> ".gz" then
87      Info(InfoWarning, 1, "Profile filenames must end in .gz to enable compression");
88    fi;
89    return ACTIVATE_PROFILING(arg[1], optRec.coverage, optRec.wallTime,
90                                      optRec.recordMem, optRec.resolution);
91end);
