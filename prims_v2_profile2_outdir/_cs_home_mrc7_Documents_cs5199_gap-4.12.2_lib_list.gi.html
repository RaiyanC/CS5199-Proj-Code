753##
754#F  IteratorList( <list> )
755##
756##  returns an iterator constructed from the list <list>,
757##  which stores the underlying list in the component `list'
758##  and the current position in the component `pos'.
759##
760##  It may happen that the underlying list is a enumerator of a domain
761##  whose size cannot be computed easily.
762##  In such cases, the methods for `IsDoneIterator' and `NextIterator'
763##  shall avoid calling `Length' for the enumerator.
764##  Therefore a special representation exist for iterators of dense immutable
765##  lists.
766##  (Note that the `Length' call is unavoidable for iterators of non-dense
767##  lists.)
768##
769BindGlobal( "IsDoneIterator_List",
770    iter -> ( iter!.pos >= iter!.len ) );
771
772BindGlobal( "NextIterator_List", function ( iter )
773    local p, l;
774    p := iter!.pos;
775    if p = iter!.len then
776        Error("<iter> is exhausted");
777    fi;
778    l := iter!.list;
779    p := p + 1;
780    while not IsBound( l[ p ] ) do
781        p := p + 1;
782    od;
783    iter!.pos := p;
784    return l[ p ];
785    end );
786
787
788BindGlobal( "NextIterator_DenseList", function ( iter )
789    local p;
790    p := iter!.pos + 1;
791    iter!.pos := p;
792    #if not IsBound( iter!.list[ iter!.pos ] ) then
793    #    Error("<iter> is exhausted");
794    #fi;
795    return iter!.list[ p ];
796    end );
797
798BindGlobal( "ShallowCopy_List",
799    iter -> rec( list := iter!.list,
800                 pos  := iter!.pos,
801                 len  := iter!.len ) );
802
803InstallGlobalFunction( IteratorList, function ( list )
804    local   iter;
805
806    iter := rec(
807                list := list,
808                pos  := 0,
809                len := Length(list),
810                IsDoneIterator := IsDoneIterator_List,
811                ShallowCopy := ShallowCopy_List );
812
813    if IsDenseList( list ) and not IsMutable( list ) then
814      iter.NextIterator := NextIterator_DenseList;
815    else
816      iter.NextIterator := NextIterator_List;
817    fi;
818
819    return IteratorByFunctions( iter );
820end );
