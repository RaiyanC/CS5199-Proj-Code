570##  <Prop Name="IsIntegralCyclotomic" Arg='obj'/>
571##
572##  <Description>
573##  A cyclotomic is called <E>integral</E> or a <E>cyclotomic integer</E>

886#m  Order( <cyc> ) . . . . . . . . . . . . . . . . .  order of an alg. number
887##
888##  If <cyc> is not a cyclotomic integer then its order is infinity.
889##  Otherwise, <cyc> is a root of unity iff its absolute value is $1$.
890##  (This follows from the more general theorem that an algebraic integer is
891##  a root of unity iff all its algebraic conjugates have absolute value $1$;
892##  note that we assume that <cyc> lies in a cyclotomic field,
893##  so the Galois group of the field extension is abelian.)
894##
895##  This method is thought for cyclotomics for which it is cheap to decide
896##  whether they are algebraic integers, and to compute the conductor;
897##  both conditions hold for internally represented cyclotomics,
898##  since they are represented w.r.t. an integral basis of the smallest
899##  possible cyclotomic field.
900##
901InstallMethod( Order,
902    "for a cyclotomic",
903    [ IsCyc ],
904    function ( cyc )
905    local n;
906
907    # Check that the argument is a root of unity.
908    if cyc = 0 then
909      Error( "argument must be nonzero" );
910    elif not IsIntegralCyclotomic( cyc )
911         or cyc * GaloisCyc( cyc, -1 ) <> 1 then
912      return infinity;
913    fi;
914
915    # Let $n$ be the conductor of `cyc'.
916    # The roots of unity in the $n$-th cyclotomic field are exactly the
917    # $n$-th roots if $n$ is even, and the $2 n$-th roots if $n$ is odd.
918    n:= Conductor( cyc );
919    if n mod 2 = 0 or cyc^n = 1 then
920      return n;
921    else
922      Assert( 1, cyc^n = -1 );
923      return 2*n;
924    fi;
925    end );
