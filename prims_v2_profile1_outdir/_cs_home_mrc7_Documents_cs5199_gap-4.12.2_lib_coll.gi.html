2832    # of non-matching stride

2848            fi;

2873        # the stuff within r2 but which it misses
2874        #

2881        fi;

2883            stride2 := r2[2]-r2[1];

2889        fi;

2891        return outs;
2892    end;

2894    #

2910        return Set(Concatenation(tounite));

2912        return goal;
