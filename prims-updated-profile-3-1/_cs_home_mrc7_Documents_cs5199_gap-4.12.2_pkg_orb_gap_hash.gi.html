520
521    ht!.accesses := ht!.accesses + 1;
522    h := HT_Hash(ht,x);
523    g := 0;
524    while IsBound(ht!.els[h]) do
525      if ht!.eqf(ht!.els[h],x) then
526          if IsBound(ht!.vals[h]) then
527              old := ht!.vals[h];
528              ht!.vals[h] := v;
529              return old;
530          else
531              ht!.vals[h] := v;
532              return true;
533          fi;
534      fi;
535      if g = 0 then
536        g := GcdInt(ht!.len,h);
537        if g = 1 then g := h; else g := 1; fi;
538      fi;
539      ht!.collisions := ht!.collisions + 1;
540      h := h+g;
541      if h>ht!.len then h := h - ht!.len; fi;
542    od;
543    return fail;
544end );

550    oldels := ht!.els;
551    oldvals := ht!.vals;
552    oldlen := ht!.len;
553
554    ht!.len := NextPrimeInt(ht!.len * 20+1);
555    Info(InfoOrb,2,"Growing tree hash table to length ",ht!.len," !!!");
556    ht!.els := EmptyPlist(ht!.len+1);
557    ht!.els[ht!.len+1] := fail;
558    ht!.vals := [];
559    if IsBound(ht!.forflatplainlists) then
560        ht!.hfd := ht!.len;
561    elif IsBound(ht!.hfbig) and IsBound(ht!.htdbig) then
562        ht!.hf := ORB_HashFunctionModWrapper;
563        ht!.hfd := [ht!.hfbig,ht!.hfdbig,ht!.len];
564    else
565        ht!.hf := ChooseHashFunction(x,ht!.len);
566        if ht!.hf = fail then
567            Error("Could not find hash function for sample object");
568            return fail;
569        fi;
570        ht!.hfd := ht!.hf.data;
571        ht!.hf := ht!.hf.func;
572    fi;
573    ht!.nr := 0;
574    ht!.collisions := 0;
575    ht!.accesses := 0;
576    # Now copy into new hash:
577    for i in [1..oldlen] do
578        if IsBound(oldels[i]) then
579            t := oldels[i];
580            if not(IsAVLTree(t)) then
581                if IsBound(oldvals[i]) then
582                    HTAdd(ht,t,oldvals[i]);
583                else
584                    HTAdd(ht,t,true);
585                fi;
586            else
587                for j in [1..Length(t)] do
588                    pos := AVLIndexFind(t,j);
589                    HTAdd(ht,AVLData(t,pos),AVLValue(t,pos));
590                od;
591            fi;
592        fi;
593    od;
594  end );

676  [IsGF2VectorRep and IsList,IsInt],
677  function(p,hashlen)
678    local bytelen;
679    bytelen := QuoInt(Length(p),8);
680    # Note that unfortunately gf2 vectors are not "clean" after their
681    # "official" length, therefore we *must not* use the last, half-used
682    # byte. This inevitably leads to collisions!
683    if bytelen <= 8 then
684        return rec( func := ORB_HashFunctionForShortGF2Vectors,
685                    data := [hashlen] );
686    else
687        return rec( func := ORB_HashFunctionForGF2Vectors,
688                    data := [hashlen,bytelen] );
689    fi;
690  end );

692InstallMethod( ChooseHashFunction, "for compressed 8bit vectors",
693  [Is8BitVectorRep and IsList,IsInt],
694  function(p,hashlen)
695    local bytelen,i,q,qq;
696    q := Q_VEC8BIT(p);
697    qq := q;

705    # Note that unfortunately 8bit vectors are not "clean" after their
706    # "official" length, therefore we *must not* use the last, half-used
707    # byte. This inevitably leads to collisions!
