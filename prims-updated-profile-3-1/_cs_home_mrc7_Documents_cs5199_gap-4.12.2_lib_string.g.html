48#C  IsString( <obj> ) . . . . . . . . . . . . . . . . . . category of strings

50##  <#GAPDoc Label="IsString">
51##  <ManSection>
52##  <Filt Name="IsString" Arg='obj'/>
53##
54##  <Description>
55##  A <E>string</E> is a dense list (see&nbsp;<Ref Filt="IsList"/>,
56##  <Ref Filt="IsDenseList"/>) of characters (see&nbsp;<Ref Filt="IsChar"/>);
57##  thus strings are always homogeneous
58##  (see&nbsp;<Ref Filt="IsHomogeneousList"/>).
59##  <P/>
60##  A string literal can either be entered as the list of characters
61##  or by writing the characters between <E>doublequotes</E> <C>"</C>.
62##  &GAP; will always output strings in the latter format.
63##  However, the input via the double quote syntax enables &GAP; to store
64##  the string in an efficient compact internal representation.
65##  See <Ref Filt="IsStringRep"/> below for more details.
66##  <P/>
67##  Each character, in particular those which cannot be typed directly from
68##  the keyboard, can also be typed in three digit octal notation, or
69##  two digit hexadecimal notation.
70##  And for some special characters (like the newline character) there is a
71##  further possibility to type them,
72##  see section <Ref Sect="Special Characters"/>.
73##  <P/>
74##  <Example><![CDATA[
75##  gap> s1 := ['H','e','l','l','o',' ','w','o','r','l','d','.'];
76##  "Hello world."
77##  gap> IsString( s1 );
78##  true
79##  gap> s2 := "Hello world.";
80##  "Hello world."
81##  gap> s1 = s2;
82##  true
83##  gap> s3 := "";  # the empty string
84##  ""
85##  gap> s3 = [];
86##  true
87##  gap> IsString( [] );
88##  true
89##  gap> IsString( "123" );  IsString( 123 );
90##  true
91##  false
92##  gap> IsString( [ '1', '2', '3' ] );
93##  true
94##  gap> IsString( [ '1', '2', , '4' ] );  # strings must be dense
95##  false
96##  gap> IsString( [ '1', '2', 3 ] );  # strings must only contain characters
