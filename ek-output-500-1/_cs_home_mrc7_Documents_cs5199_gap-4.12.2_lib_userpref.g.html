33##  <Description>
34##  
35##  Some aspects of the behaviour of &GAP; can be customized by the user via
36##  <Emph>user preferences</Emph>.  Examples include  the way  help sections
37##  are displayed or the use of colors in the terminal. <P/>
38##  
39##  User preferences are  specified via a pair of strings,  the first is the
40##  (case insensitive) name of a package (or <C>"GAP"</C> for the core &GAP;
41##  library) and the second is some arbitrary case sensitive string. <P/>
42##  
43##  User   preferences  can   be  set   to  some   <A>value</A>  with   <Ref
44##  Func="SetUserPreference"/>. The  current value of a  user preference can
45##  be found with <Ref Func="UserPreference"/>. In both cases, if no package
46##  name is  given the default  <C>"GAP"</C> is  used. If a  user preference
47##  is  not  known or  not  set  then <Ref  Func="UserPreference"/>  returns
48##  <K>fail</K>. <P/>
49##  
50##  The function <Ref Func="ShowUserPreferences"/> with no argument shows in
51##  a  pager  an  overview  of  all known  user  preferences  together  with
52##  some  explanation  and  the  current  value.  If  one  or  more  strings
53##  <A>package1</A>, ... are given then  only the user preferences for these
54##  packages are shown. <P/>
55##  
56##  The easiest way to  make use of user preferences is  probably to use the
57##  function <Ref  Func="WriteGapIniFile"/>, usually without  argument. This
58##  function creates a file <F>gap.ini</F>  in your user specific &GAP; root
59##  directory (<C>GAPInfo.UserGapRoot</C>).  If such  a file  already exists
60##  the function  will make a  backup of it  first. This newly  created file
61##  contains  descriptions of  all  known user  preferences  and also  calls
62##  of  <Ref Func="SetUserPreference"/>  for  those  user preferences  which
63##  currently do not  have their default value. You can  then edit that file
64##  to customize (further)  the user preferences for  future &GAP; sessions.
65##  <P/>
66##  
67##  Should a  later version  of &GAP;  or some  packages introduce  new user
68##  preferences then you can  call <Ref Func="WriteGapIniFile"/> again since
69##  it  will set  the previously  known  user preferences  to their  current
70##  values. <P/>
71##  
72##  Optionally,  a  different  directory for  the  resulting  <F>gap.ini</F>
73##  file    can   be    specified   as    argument   <A>dir</A>    to   <Ref
74##  Func="WriteGapIniFile"/>. Another optional argument is the boolean value
75##  <K>true</K>, if this  is given, the settings of all  user preferences in
76##  the current session are ignored. <P/>
77##  
78##  Note that  your <F>gap.ini</F> file is  read by &GAP; very  early during
79##  its startup process. A consequence  is that the <A>value</A> argument in
80##  a call of <Ref Func="SetUserPreference"/>  must be some very basic &GAP;
81##  object, usually a boolean, a number, a  string or a list of those. A few
82##  user  preferences support  more complicated  settings. For  example, the
83##  user  preference <C>"UseColorPrompt"</C>  admits a  record as  its value
84##  whose components are available  only after the <Package>GAPDoc</Package>
85##  package has been loaded, see&nbsp;<Ref Func="ColorPrompt"/>. If you want
86##  to specify such a complicated value, then move the corresponding call of
87##  <Ref Func="SetUserPreference"/> from your  <F>gap.ini</F> file into your
88##  <F>gaprc</F>  file (also  in the  directory <C>GAPInfo.UserGapRoot</C>).
89##  This file is read much later. <P/>
90##  
91##  <Example>
92##  gap> SetUserPreference( "Pager", "less" );
93##  gap> SetUserPreference("PagerOptions",
94##  >                      [ "-f", "-r", "-a", "-i", "-M", "-j2" ] );
95##  gap> UserPreference("Pager");
96##  "less"

101##  available on  your system. The  last line displays the  current setting.

110##  <Description>
111##  
112##  This  function can  be used  (also in  packages) to  introduce new  user
113##  preferences. It declares  a user preference, determines  a default value
114##  and contains documentation  of the user preference.  After declaration a
115##  user preference will be shown with <Ref Func="ShowUserPreferences"/> and
116##  <Ref Func="WriteGapIniFile"/>. <P/>
117##  
118##  When  this  declaration  is  evaluated  it  is  checked,  if  this  user
119##  preference is  already set in the  current session. If not  the value of
120##  the user  preference is set to  its default. (Do not  use <K>fail</K> as
121##  default  value  since this  indicated  that  a  user preference  is  not
122##  set.)<P/>
123##  
124##  The argument <A>record</A> of <Ref Func="DeclareUserPreference"/> must be
125##  a record with the following components.
126##  <P/>
127##  <List>
128##  <Mark><C>name</C></Mark>
129##  <Item>
130##    a string or a list of strings, the latter meaning several preferences
131##    which belong together,
132##  </Item>
133##  <Mark><C>description</C></Mark>
134##  <Item>
135##    a list of strings describing the preference(s), one string for each
136##    paragraph;
137##    if several preferences are declared together then the description
138##    refers to all of them,
139##  </Item>
140##  <Mark><C>default</C></Mark>
141##  <Item>
142##    the default value that is used,
143##    or a function without arguments that computes this default value;
144##    if several preferences are declared together then the value of this
145##    component must be the list of default values for the individual
146##    preferences.
147##  </Item>
148##  </List>
149##  <P/>
150##  The following components of <A>record</A> are optional.
151##  <P/>
152##  <List>
153##  <Mark><C>check</C></Mark>
154##  <Item>
155##    a function that takes a value as its argument and returns either
156##    <K>true</K> or <K>false</K>, depending on whether the given value
157##    is admissible for this preference;
158##    if several preferences are declared together then the number of
159##    arguments of the function must equal the length of the <C>name</C>
160##    list,
161##  </Item>
162##  <Mark><C>values</C></Mark>
163##  <Item>
164##    the list of admissible values, or a function without arguments
165##    that returns this list,
166##  </Item>
167##  <Mark><C>multi</C></Mark>
168##  <Item>
169##    <K>true</K> or <K>false</K>, depending on whether one may choose
170##    several values from the given list or just one;
171##    needed (and useful only) if the <C>values</C> component is present,
172##  </Item>
173##  <Mark><C>package</C></Mark>
174##  <Item>
175##    the name of the &GAP; package to which the preference is assigned;
176##    if the declaration happens inside a file that belongs to this package
177##    then the value of this component is computed,
178##    using <C>GAPInfo.PackageCurrent</C>;
179##    otherwise, the default value for <C>package</C> is <C>"GAP"</C>,
180##  </Item>
181##  <Mark><C>omitFromGapIniFile</C></Mark>
182##  <Item>
183##    if the value is <K>true</K> then this user preference is ignored by
184##    <Ref Func="WriteGapIniFile"/>.
185##  </Item>
186##  </List>
187##  <P/>
188##  <Example><![CDATA[
189##  gap> UserPreference( "MyFavouritePrime" );
190##  fail
191##  gap> DeclareUserPreference( rec(
192##  >        name:= "MyFavouritePrime",
193##  >        description:= [ "is not used, serves as an example" ],
194##  >        default:= 2,
195##  >        omitFromGapIniFile:= true ) );
196##  gap> UserPreference( "MyFavouritePrime" );
197##  2
198##  gap> SetUserPreference( "MyFavouritePrime", 17 );
199##  gap> UserPreference( "MyFavouritePrime" );

680BindGlobal( "WriteGapIniFile", function( arg )
681  local ignorecurrent, f, df, ret, target, str, res;
682  # check if current settings should be used
683  if true in arg then 
684    ignorecurrent := true;
685  else 
686    ignorecurrent := false;
687  fi;
688  # check if target directory is given as string or directory object
689  f := First(arg, IsString);
690  if f <> fail then
691    df := Directory(f);
692  else
693    df := First(arg, IsDirectory);
694    if df <> fail then
695      f := df![1];
696    fi;
697  fi;
698  # otherwise use users GAPInfo.UserGapRoot
699  if f = fail then
700    if GAPInfo.UserGapRoot = fail then
701      Error("Your system does not support a user specific default root path.\n\
702Please specify directory to write gap.ini file.");
703      return fail;
704    else
705      f := GAPInfo.UserGapRoot;
706      df := Directory(f);
707    fi;
708  fi;
709  # maybe create GAPInfo.UserGapRoot
710  if not IsDirectoryPath(f) then
711    ret := CreateDir(f);
712    if ret = fail then
713      Error("Cannot create directory ",f,"\nError message: ",
714            LastSystemError().message,"\n");
715      return fail;
716    fi;
717  fi;
718  # name of target file
719  target := Filename(df, "gap.ini");
720  # if target exists copy it to <target>.bak
721  if IsExistingFile(target) then
722    str := StringFile(target);
723    if str = fail then
724      Error("Cannot read existing file ",target,".\n");
725      return fail;
726    fi;
727    ret := FileString(Concatenation(target, ".bak"), str);
728    if ret = fail then
729      Error("Cannot write backup file ",target,".bak.\nError message: ",
730            LastSystemError().message,".\n");
731      return fail;
732    else
733      Info(InfoWarning, 1, "Copied existing gap.ini to ", target, ".bak");
734    fi;
735  fi;
736  # content of resulting file as string
737  res := StringUserPreferences(ignorecurrent);
738  if ARCH_IS_WINDOWS() then
739    # use DOS/Windows style line breaks
740    res := ReplacedString(res, "\n", "\r\n");
741  fi;
742  # finally write the gap.ini file
743  ret := FileString(target, res);
744  if ret = fail then
745    Error("Cannot write target file ",target,".\nError message: ",
746          LastSystemError().message,".\n");
747    return fail;
748  fi;
749  Info(InfoWarning, 1, "File ", target, " successfully written.");
750  return true;
751end );
